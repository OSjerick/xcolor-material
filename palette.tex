\newcommand*{\materialpalettes}[2][]{%
	\pgfkeys{/palette, default, #1}
	\def\mycolor{#2}
	\begin{tikzpicture}
		\tikzset{colorbox/.style={rectangle, minimum width=\paletteWidth, minimum height=\paletteHeight, node distance=\paletteSep, outer sep=0pt}}
		\tikzset{descript/.style = {font=\paletteDFont, color=\paletteTextColor, inner sep=1em, align=left}}
		\tikzset{colors/.style = {font=\paletteFont, inner sep=1em, align=left}}
		\tikzset{every node/.style = {colors}}
		
		\node[colorbox, fill=\mycolor\paletteColorSep\palettePrimary, minimum height=\paletteStartHeight, alias=last] {};
		\node[descript, anchor=north west] at (last.north west) {\mycolor};
		\node[descript, anchor=south west] at (last.south west) {\palettePrimary};
		\node[descript, anchor=south east, align=right] at (last.south east) {\texttt{\extractHEXColor{\mycolor\paletteColorSep\palettePrimary}}};
		
		\def\auxcolor{\paletteFTextColor}
		\def\first{1}
		\foreach \x [count=\i] in \paletteVariant {
			\ifx\i\first \node[node distance=\paletteStartSep, inner sep=0pt, below=of last, alias=last] {};\else\fi 
			\ifx\x\paletteChange \gdef\auxcolor{\paletteSTextColor}\else\fi
			\node[colorbox, fill=\mycolor\paletteColorSep\x, below=of last, alias=last] {};
			\node[anchor=west, color=\auxcolor] at (last.west) {\x};
			\node[anchor=east, color=\auxcolor, align=right] at (last.east) {\texttt{\extractHEXColor{\mycolor\paletteColorSep\x}}};
		}
		
		\def\auxcolor{\paletteFTextColor}
		\foreach \x [count=\i] in \paletteAddVariant {
			\ifx\i\first \node[node distance=\paletteStartSep, inner sep=0pt, below=of last, alias=last] {};\else\fi 
			\ifx\x\paletteAddChange \gdef\auxcolor{\paletteSTextColor}\else\fi
			\node[colorbox, fill=\mycolor\paletteColorSep\x, below=of last, alias=last] {};
			\node[anchor=west, color=\auxcolor] at (last.west) {\x};
			\node[anchor=east, color=\auxcolor, align=right] at (last.east) {\texttt{\extractHEXColor{\mycolor\paletteColorSep\x}}};
		}
	\end{tikzpicture}
}
