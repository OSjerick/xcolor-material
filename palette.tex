\usetikzlibrary{positioning}
\newcommand*{\materialpalettes}[2][]{%
	\pgfkeys{/palette, default, #1}
	\def\mycolor{#2}
	\begin{tikzpicture}
		\tikzset{colorbox/.style={rectangle, minimum width=\paletteWidth, minimum height=\paletteHeight, node distance=\paletteSep, outer sep=0pt}}
		\tikzset{font=\sffamily\normalsize, color=white}
		
		\node[colorbox, fill=\mycolor\paletteColorSep\palettePrimary, minimum height=\paletteStartHeight, alias=last] {};
		\node[anchor=north west, inner sep=1em] at (last.north west) {\mycolor};
		\node[anchor=south west, inner sep=1em] at (last.south west) {\palettePrimary};
		\node[anchor=south east, inner sep=1em] at (last.south east) {\texttt{\#FF0000}};
		
		\node[node distance=\paletteStartSep, inner sep=0pt, below=of last, alias=last] {};
		\foreach \x in \paletteVariant {
			\node[colorbox, fill=\mycolor\paletteColorSep\x, below=of last, alias=last] {};
			\node[anchor=west, inner sep=1em] at (last.west) {\x};
			\node[anchor=east, inner sep=1em] at (last.east) {\texttt{\#FF0000}};
		}
		
		\node[node distance=\paletteStartSep, inner sep=0pt, below=of last, alias=last] {};
		\foreach \x in \paletteAddVariant {
			\node[colorbox, fill=\mycolor\paletteColorSep\x, below=of last, alias=last] {};
			\node[anchor=west, inner sep=1em] at (last.west) {\x};
			\node[anchor=east, inner sep=1em] at (last.east) {\texttt{\#FF0000}};
		}
	\end{tikzpicture}
}
